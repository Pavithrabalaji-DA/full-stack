    <script>
        function displayBioData() {
            var name = document.getElementById("name").value;
            var age = document.getElementById("age").value;
            var course = document.getElementById("course").value;

            if (name && age && course) {
                var bioData = "<strong>Name:</strong> " + name + "<br>" +
                              "<strong>Age:</strong> " + age + "<br>" +
                              "<strong>Course:</strong> " + course;

                document.getElementById("bioData").innerHTML = "<h2>Student Bio Data</h2>" + bioData;
            } else {
                document.getElementById("bioData").innerHTML = "<p style='color: red;'>Please fill out all fields.</p>";
            }
        }
    </script>
